        const form = document.getElementById('upload-form');
        const resumesInput = document.getElementById('resumes');
        const resumeFilenamesSpan = document.getElementById('resume-filenames');
        const resultsContainer = document.getElementById('results-container');
        const loader = document.getElementById('loader');
        const generateJdBtn = document.getElementById('generate-jd-btn');
        const jdTextarea = document.getElementById('job_description_text');
        const notificationArea = document.getElementById('notification-area');

        function showNotification(message, isError = false) {
            const notif = document.createElement('div');
            notif.className = `notification ${isError ? 'bg-red-600' : 'bg-indigo-600'}`;
            notif.textContent = message;
            notificationArea.appendChild(notif);
            setTimeout(() => notif.classList.add('show'), 10);
            setTimeout(() => {
                notif.classList.remove('show');
                setTimeout(() => notif.remove(), 300);
            }, 3000);
        }

        resumesInput.addEventListener('change', () => { resumeFilenamesSpan.textContent = resumesInput.files.length > 0 ? `${resumesInput.files.length} resume(s) selected` : ''; });

        generateJdBtn.addEventListener('click', async () => {
            const notes = jdTextarea.value;
            if (!notes.trim()) {
                showNotification('Please enter some notes to generate a description.', true);
                return;
            }
            generateJdBtn.textContent = 'Generating...';
            generateJdBtn.disabled = true;
            try {
                const response = await fetch('/generate_jd', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ notes: notes })
                });
                const data = await response.json();
                if (response.ok) {
                    jdTextarea.value = data.job_description;
                } else {
                    showNotification(data.error || 'An unknown error occurred.', true);
                }
            } catch (error) {
                console.error("JD Generation Error:", error);
                showNotification("Failed to connect to the server for JD generation.", true);
            } finally {
                generateJdBtn.textContent = '✨ Generate with AI';
                generateJdBtn.disabled = false;
            }
        });

        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            loader.classList.remove('hidden');
            resultsContainer.innerHTML = '';
            resultsContainer.appendChild(loader);
            const formData = new FormData(form);
            try {
                const response = await fetch('/process', { method: 'POST', body: formData });
                // We will check for the content type to see if we got JSON
                const contentType = response.headers.get("content-type");
                if (contentType && contentType.indexOf("application/json") !== -1) {
                    const results = await response.json();
                    displayResults(results);
                } else {
                    const text = await response.text();
                    throw new Error(`Server returned non-JSON response: ${text}`);
                }
            } catch (error) {
                console.error('Processing error:', error);
                displayError('An error occurred during processing. Please check the browser console and terminal for details.');
            } finally {
                loader.classList.add('hidden');
            }
        });

        function displayResults(results) {
            resultsContainer.innerHTML = '';
            if (!results || results.length === 0) {
                resultsContainer.innerHTML = '<p class="text-center text-gray-400">No results to display.</p>';
                return;
            }
            results.forEach(result => {
                const card = createResultCard(result);
                if (card) resultsContainer.appendChild(card);
            });
        }

        function displayError(message) {
            resultsContainer.innerHTML = `<div class="bg-red-900/50 border border-red-700 text-red-300 p-4 rounded-lg" role="alert"><p class="font-bold">Error</p><p>${message}</p></div>`;
        }
        
        // --- FINAL, ROBUST VERSION OF createResultCard ---
        function createResultCard(result) {
            if (!result?.state?.screening_results) {
                console.error("Invalid result object from backend:", result);
                return null;
            }
            const { filename, state } = result;
            const { screening_results = {}, drafted_email, final_status } = state;
            
            const candidateName = screening_results.candidateName || "N/A";
            const candidateEmail = screening_results.candidateEmail || "N/A";
            const matchScore = screening_results.matchScore || 0;
            const summary = screening_results.summary || "No summary available.";
            
            const card = document.createElement('div');
            card.className = 'bg-gray-800/50 backdrop-blur-sm p-6 rounded-2xl border';
            const isGoodMatch = matchScore >= 70;
            const borderColorClass = isGoodMatch ? 'gradient-border-good border-2' : 'gradient-border-bad border-2';
            card.classList.add(borderColorClass);
            
            let statusMessage = '';
            if (final_status) {
                statusMessage = final_status;
            } else if (drafted_email?.body) {
                statusMessage = 'Email sent successfully.';
            } else if (isGoodMatch && (candidateEmail === 'N/A' || !candidateEmail)) {
                statusMessage = 'Candidate is a good match, but no email address was found in the resume.';
            } else {
                statusMessage = 'Candidate did not meet the minimum score for an invitation.';
            }

            const draftedEmailBody = drafted_email?.body || '';

            let emailSection = `<div class="mt-4 pt-4 border-t border-gray-700"><p class="text-sm font-medium text-gray-400">Status: <span class="font-semibold ${isGoodMatch ? 'text-indigo-400' : 'text-amber-400'}">${statusMessage}</span></p></div>`;
            if (draftedEmailBody) {
                 emailSection += `<div class="mt-4 p-4 bg-gray-900 rounded-lg text-sm text-gray-300 whitespace-pre-wrap border border-gray-700"><strong>Sent Email:</strong><br><br>${draftedEmailBody.replace(/\n/g, '<br>')}</div>`;
            }

            card.innerHTML = `
                <div class="flex justify-between items-start gap-4">
                    <div>
                        <h3 class="text-xl font-bold text-white">${candidateName}</h3>
                        <p class="text-sm text-gray-400">${candidateEmail} (${filename})</p>
                    </div>
                    <div class="text-right flex-shrink-0">
                        <p class="text-sm text-gray-400">Match Score</p>

                    </div>
                </div>

                    <div class="w-full bg-gray-700 rounded-full h-2.5">
                        <div class="h-2.5 rounded-full ${isGoodMatch ? 'gradient-bg' : 'bg-amber-500'}" style="width: ${matchScore}%"></div>

                    <h4 class="font-semibold text-gray-200">AI Summary:</h4>
                    <p class="mt-1 text-sm text-gray-400">${summary}</p>
                </div>
                ${emailSection}
            `;
            return card;
        }
